def proceedSubmit
  
  case @method
  when "facebook"
   
    if !checkUser
      
    
        unless @mysqlHelper.insertUser(@userData['email'], 
                            @userData['password'],
                            @userData['last_name'], 
                            @userData['first_name'], 
                            @userData['birthdate'], 
                            @userData['uuid'],
                            @userData['facebook_id'], 
                            @userData['fb_access_token'], 
                            @userData['fb_access_token_expiration'])
                          then
                          ERROR_USER_NOT_REGISTERED
        else
          Dir.chdir(RESSOURCE_PATH+"#{@userData['uuid']}")
          SUCCESS_USER_REGISTERED
   end
   
    else
      ERROR_USER_EXIST
    end
    
  when "normal"
    
    if !checkUser
    unless @mysqlHelper.insertUser(@userData['email'], 
                            @userData['password'],
                            @userData['last_name'], 
                            @userData['first_name'],
                            nil,
                            @userData['uuid'],
                            nil, 
                            nil, 
                            nil) 
                          then
                          ERROR_USER_NOT_REGISTERED
    else
      Dir.mkdir(RESSOURCE_PATH+"#{@userData['uuid']}")
      SUCCESS_USER_REGISTERED
    end
    else
      ERROR_USER_EXIST
    end
    
  else
    ERROR_UNKNOWN_METHOD
  end
    
end
